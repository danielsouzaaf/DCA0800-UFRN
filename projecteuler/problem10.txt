#include <stdlib.h>
#include <math.h>

int checkPrime(int n)
{
	int i;
	if (n <= 1) return 0;
	for (i = 2; i*i <= n; i++)
	{
		if (n % i == 0) return 0;
	}
	return 1;
}

int main (void)
{
	
	const int limit = 2000000;
	long int soma;
	int i;
	soma = 0;
	for (i = 2; i < limit; i++)
	{
		if (checkPrime(i) == 1) soma += i;
	}
	
	printf("soma = %li", soma);
	

	return 0;
    
}
