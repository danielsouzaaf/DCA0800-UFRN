#include <stdio.h>
#include <stdlib.h>
#include <math.h>

float desvioPadrao(float x[], float media, int n)
{
	float desviopadrao;
	desviopadrao = 0.0;
	int j;
	for (j = 0; j < n; j++)
	{
		desviopadrao += pow(x[j] - media, 2);
	}
	return sqrt(desviopadrao / (n - 1));
}
